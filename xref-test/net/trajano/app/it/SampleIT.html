16  /**
17   * Tests the Java EE web application components.
18   */
19  public class SampleIT {
20      /**
21       * Integration test context.
22       */
23      private IntegrationTestContext context;
24  
25      /**
26       * Terminates the {@link WebDriver}.
27       */
28      @After
29      public void closeContext() {
30          context.close();
31      }
32  
33      @Before
34      public void setupContext() throws IOException {
35          context = new IntegrationTestContext();
36      }
37  
38      /**
39       * Tests a REST API call using Angular to make the call to the REST API.
40       */
41      @Test
42      public void testRest() throws Exception {
43          context.authenticate();
44          context.open();
45          context.getWait().until(
46                  textToBePresentInElementLocated(By.id("bean-message"),
47                  "Hello JAX-RS"));
48          context.getWait().until(
49                  textToBePresentInElementLocated(By.id("cdi-message"),
50                          "Hello CDI"));
51          context.getWait().until(
52                  textToBePresentInElementLocated(By.id("jpa-message"),
53                          "Hello JPA"));
54  
55          assertThat(context.getDriver().findElement(By.id("bean-attribute"))
56                  .getText(), is("bean"));
57          assertThat(context.getDriver().findElement(By.id("bean-message"))
58                  .getText(), is("Hello JAX-RS"));
59          assertThat(context.getDriver().findElement(By.id("cdi-message"))
60                  .getText(), is("Hello CDI"));
61          assertThat(context.getDriver().findElement(By.id("jpa-message"))
62                  .getText(), is("Hello JPA"));
63      }
64  
65      /**
66       * Tests the servlet call.
67       */
68      @Test
69      public void testServlet() {
70          context.openServlet();
71          assertThat(context.getDriver().findElement(By.id("text")).getText(),
72                  startsWith("Hello servlet on "));
73      }
74  }
