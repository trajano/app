3   import java.util.logging.Logger;

5   import javax.enterprise.context.ApplicationScoped;
6   import javax.interceptor.AroundInvoke;
7   import javax.interceptor.Interceptor;
8   import javax.interceptor.InvocationContext;
9   
10  import net.trajano.app.BusinessMethod;
11  
12  /**
13   * @author Archimedes
14   */
15  @BusinessMethod
16  @Interceptor
17  @ApplicationScoped
18  public class LoggingInterceptor {
19      /**
20       * Logger.
21       */
22      private static final Logger LOG = Logger.getLogger("net.trajano.app");
23  
24      /**
25       * Logs an invocation event.
26       * 
27       * @param ctx
28       *            context
29       * @return return value
30       * @throws Throwable
31       */
32      @AroundInvoke
33      public Object logInvocation(final InvocationContext ctx) throws Throwable {
34          LOG.warning("around invoke " + ctx.getMethod());
35          try {
36              return ctx.proceed();
37          } catch (final Throwable e) {
38              LOG.severe("Error calling ctx.proceed in " + ctx.getMethod());
39              throw e;
40          }
41      }
42  }
