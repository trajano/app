3   import java.io.IOException;
4   
5   import javax.websocket.OnMessage;
6   import javax.websocket.Session;
7   import javax.websocket.server.ServerEndpoint;
8   
9   /**
10   * Web socket endpoint.
11   * 
12   * @author Archimedes
13   */
14  @ServerEndpoint("/echo")
15  public class EchoEndpoint {
16      /**
17       * Sends back the original message.
18       * 
19       * @param session
20       *            web socket session
21       * @param message
22       *            the message
23       * @throws IOException
24       */
25      @OnMessage
26      public void onMessage(final Session session, final String message)
27              throws IOException {
28          session.getBasicRemote().sendText(message);
29      }
30  }
